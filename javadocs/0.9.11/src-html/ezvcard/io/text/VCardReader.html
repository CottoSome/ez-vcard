065/**
066 * <p>
067 * Parses {@link VCard} objects from a plain-text vCard data stream.
068 * </p>
069 * <p>
070 * <b>Example:</b>
071 * </p>
072 * 
073 * <pre class="brush:java">
074 * File file = new File("vcards.vcf");
075 * VCardReader reader = null;
076 * try {
077 *   reader = new VCardReader(file);
078 *   VCard vcard;
079 *   while ((vcard = reader.readNext()) != null){
080 *     ...
081 *   }
082 * } finally {
083 *   if (reader != null) reader.close();
084 * }
085 * </pre>
086 * @author Michael Angstadt
087 * @see <a href="http://www.imc.org/pdi/vcard-21.rtf">vCard 2.1</a>
088 * @see <a href="http://tools.ietf.org/html/rfc2426">RFC 2426 (3.0)</a>
089 * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 (4.0)</a>
090 */
091public class VCardReader extends StreamReader {
092        private final VCardRawReader reader;
093        private Charset defaultQuotedPrintableCharset;

124        public VCardReader(Reader reader) {
125                this.reader = new VCardRawReader(reader);
126                defaultQuotedPrintableCharset = this.reader.getEncoding();

128                        defaultQuotedPrintableCharset = Charset.defaultCharset();

200        @Override
201        protected VCard _readNext() throws IOException {
202                /*
203                 * Although this almost never happens, vCards can be nested inside of
204                 * each other (see: AGENT property).
205                 */
206                VCard root = null;
207                VCardStack stack = new VCardStack();
208                EmbeddedVCardException embeddedVCardException = null;
209
210                while (true) {
211                        //read next line
212                        VCardRawLine line;
213                        try {
214                                line = reader.readLine();
215                        } catch (VCardParseException e) {
216                                if (!stack.isEmpty()) {
217                                        warnings.add(e.getLineNumber(), null, 27, e.getMessage(), e.getLine());
218                                }
219                                continue;
220                        }
221
222                        //EOF
223                        if (line == null) {
224                                break;
225                        }
226
227                        //handle BEGIN:VCARD
228                        if ("BEGIN".equalsIgnoreCase(line.getName()) && "VCARD".equalsIgnoreCase(line.getValue())) {
229                                VCard vcard = new VCard();
230                                vcard.setVersion(reader.getVersion());
231                                stack.push(vcard);
232
233                                if (root == null) {
234                                        root = vcard;
235                                }
236
237                                if (embeddedVCardException != null) {
238                                        embeddedVCardException.injectVCard(vcard);
239                                        embeddedVCardException = null;
240                                }
241
242                                continue;
243                        }
244
245                        if (stack.isEmpty()) {
246                                //BEGIN component hasn't been encountered yet, so skip this line
247                                continue;
248                        }
249
250                        //handle VERSION property
251                        if ("VERSION".equalsIgnoreCase(line.getName())) {
252                                stack.peek().vcard.setVersion(reader.getVersion());
253                                continue;
254                        }
255
256                        //handle END:VCARD
257                        if ("END".equalsIgnoreCase(line.getName()) && "VCARD".equalsIgnoreCase(line.getValue())) {
258                                VCardStack.Item item = stack.pop();
259                                assignLabels(item.vcard, item.labels);
260
261                                if (stack.isEmpty()) {
262                                        //done reading the vCard
263                                        break;
264                                }
265                                continue;
266                        }
267
268                        //handle property
269                        {
270                                String group = line.getGroup();
271                                VCardParameters parameters = line.getParameters();
272                                String name = line.getName();
273                                String value = line.getValue();
274
275                                if (embeddedVCardException != null) {
276                                        //the next property was supposed to be the start of a nested vCard, but it wasn't
277                                        embeddedVCardException.injectVCard(null);
278                                        embeddedVCardException = null;
279                                }
280
281                                VCard curVCard = stack.peek().vcard;
282                                VCardVersion version = curVCard.getVersion();
283
284                                //sanitize the parameters
285                                processNamelessParameters(parameters);
286                                processQuotedMultivaluedTypeParams(parameters);
287
288                                //decode property value from quoted-printable
289                                try {
290                                        value = decodeQuotedPrintableValue(name, parameters, value);
291                                } catch (DecoderException e) {
292                                        warnings.add(reader.getLineNumber(), name, 38, e.getMessage());
293                                }
294
295                                //get the scribe
296                                VCardPropertyScribe<? extends VCardProperty> scribe = index.getPropertyScribe(name);
297                                if (scribe == null) {
298                                        scribe = new RawPropertyScribe(name);
299                                }
300
301                                //get the data type (VALUE parameter)
302                                VCardDataType dataType = parameters.getValue();
303                                parameters.setValue(null);
304                                if (dataType == null) {
305                                        //use the property's default data type if there is no VALUE parameter
306                                        dataType = scribe.defaultDataType(version);
307                                }
308
309                                VCardProperty property;
310                                try {
311                                        Result<? extends VCardProperty> result = scribe.parseText(value, dataType, version, parameters);
312
313                                        for (String warning : result.getWarnings()) {
314                                                warnings.add(reader.getLineNumber(), name, warning);
315                                        }
316
317                                        property = result.getProperty();
318                                        property.setGroup(group);
319
320                                        if (property instanceof Label) {
321                                                /*
322                                                 * LABEL properties must be treated specially so they
323                                                 * can be matched up with the ADR properties that they
324                                                 * belong to.
325                                                 */
326                                                Label label = (Label) property;
327                                                stack.peek().labels.add(label);
328                                        } else {
329                                                curVCard.addProperty(property);
330                                        }
331                                } catch (SkipMeException e) {
332                                        warnings.add(reader.getLineNumber(), name, 22, e.getMessage());
333                                } catch (CannotParseException e) {
334                                        warnings.add(reader.getLineNumber(), name, 25, value, e.getMessage());
335                                        property = new RawPropertyScribe(name).parseText(value, dataType, version, parameters).getProperty();
336                                        property.setGroup(group);
337                                        curVCard.addProperty(property);
338                                } catch (EmbeddedVCardException e) {
339                                        //parse an embedded vCard (i.e. the AGENT type)
340                                        property = e.getProperty();
341
342                                        if (value.length() == 0 || version == VCardVersion.V2_1) {
343                                                //a nested vCard is expected to be next (2.1 style)
344                                                embeddedVCardException = e;
345                                        } else {
346                                                //the property value should be an embedded vCard (3.0 style)
347                                                value = VCardPropertyScribe.unescape(value);
348
349                                                VCardReader agentReader = new VCardReader(value);
350                                                agentReader.setScribeIndex(index);
351                                                try {
352                                                        VCard nestedVCard = agentReader.readNext();
353                                                        if (nestedVCard != null) {
354                                                                e.injectVCard(nestedVCard);
355                                                        }
356                                                } catch (IOException e2) {
357                                                        //shouldn't be thrown because we're reading from a string
358                                                } finally {
359                                                        for (String w : agentReader.getWarnings()) {
360                                                                warnings.add(reader.getLineNumber(), name, 26, w);
361                                                        }
362                                                        IOUtils.closeQuietly(agentReader);
363                                                }
364                                        }
365
366                                        curVCard.addProperty(property);
367                                }
368                        }
369                }
370
371                return root;
372        }

374        /**
375         * Assigns names to all nameless parameters. v3.0 and v4.0 requires all
376         * parameters to have names, but v2.1 does not.
377         * @param parameters the parameters
378         */
379        private void processNamelessParameters(VCardParameters parameters) {
380                List<String> namelessParamValues = parameters.removeAll(null);
381                for (String paramValue : namelessParamValues) {
382                        String paramName = guessParameterName(paramValue);
383                        parameters.put(paramName, paramValue);
384                }
385        }
386
387        /**
388         * Makes a guess as to what a parameter value's name should be.
389         * @param value the parameter value
390         * @return the guessed name
391         */
392        private String guessParameterName(String value) {
393                if (VCardDataType.find(value) != null) {
394                        return VCardParameters.VALUE;
395                }
396
397                if (Encoding.find(value) != null) {
398                        return VCardParameters.ENCODING;
399                }
400
401                //otherwise, assume it's a TYPE
402                return VCardParameters.TYPE;
403        }

405        /**
406         * <p>
407         * Accounts for multi-valued TYPE parameters being enclosed entirely in
408         * double quotes (for example: ADR;TYPE="home,work").
409         * </p>
410         * <p>
411         * Many examples throughout the 4.0 specs show TYPE parameters being encoded
412         * in this way. This conflicts with the ABNF and is noted in the errata.
413         * This method will parse these incorrectly-formatted TYPE parameters as if
414         * they were multi-valued, even though, technically, they are not.
415         * </p>
416         * @param parameters the parameters
417         */
418        private void processQuotedMultivaluedTypeParams(VCardParameters parameters) {
419                List<String> types = parameters.getTypes();
420                String valueWithComma = null;
421                for (String value : types) {
422                        if (value.indexOf(',') >= 0) {
423                                valueWithComma = value;
424                                break;
425                        }
426                }
427                if (valueWithComma == null) {
428                        return;
429                }
430
431                types.clear();
432                int prev = -1, cur;
433                while ((cur = valueWithComma.indexOf(',', prev + 1)) >= 0) {
434                        types.add(valueWithComma.substring(prev + 1, cur));
435                        prev = cur;
436                }
437                types.add(valueWithComma.substring(prev + 1));
438        }

440        /**
441         * Checks to see if a property's value is encoded in quoted-printable
442         * encoding and decodes it if it is.
443         * @param name the property name
444         * @param parameters the property parameters
445         * @param value the property value (may or may not be encoded in
446         * quoted-printable
447         * @return the decoded property value or the untouched property value if it
448         * is not encoded in quoted-printable encoding
449         * @throws DecoderException if the value couldn't be decoded
450         */
451        private String decodeQuotedPrintableValue(String name, VCardParameters parameters, String value) throws DecoderException {
452                if (parameters.getEncoding() != Encoding.QUOTED_PRINTABLE) {
453                        //the property value is not encoded in quoted-printable encoding
454                        return value;
455                }
456
457                //remove the encoding parameter
458                parameters.setEncoding(null);
459
460                //determine the character set
461                Charset charset = null;
462                String charsetStr = parameters.getCharset();
463                if (charsetStr == null) {
464                        charset = defaultQuotedPrintableCharset;
465                } else {
466                        try {
467                                charset = Charset.forName(charsetStr);
468                        } catch (IllegalCharsetNameException e) {
469                                //bad charset name
470                        } catch (UnsupportedCharsetException e) {
471                                //bad charset name
472                        }
473
474                        if (charset == null) {
475                                charset = defaultQuotedPrintableCharset;
476
477                                //the given charset was invalid, so add a warning
478                                warnings.add(reader.getLineNumber(), name, 23, charsetStr, charset.name());
479                        }
480                }
481
482                QuotedPrintableCodec codec = new QuotedPrintableCodec(charset.name());
483                return codec.decode(value);
484        }

486        /**
487         * Keeps track of the hierarchy of nested vCards.
488         */
489        private static class VCardStack {
490                private final List<Item> stack = new ArrayList<Item>();
491
492                /**
493                 * Adds a vCard to the stack.
494                 * @param vcard the vcard to add
495                 */
496                public void push(VCard vcard) {
497                        stack.add(new Item(vcard, new ArrayList<Label>()));
498                }
499
500                /**
501                 * Removes the top item from the stack and returns it.
502                 * @return the last item or null if the stack is empty
503                 */
504                public Item pop() {
505                        return isEmpty() ? null : stack.remove(stack.size() - 1);
506                }
507
508                /**
509                 * Gets the top item of the stack.
510                 * @return the top item
511                 */
512                public Item peek() {
513                        return isEmpty() ? null : stack.get(stack.size() - 1);
514                }
515
516                /**
517                 * Determines if the stack is empty.
518                 * @return true if it's empty, false if not
519                 */
520                public boolean isEmpty() {
521                        return stack.isEmpty();
522                }
523
524                private static class Item {
525                        public final VCard vcard;
526                        public final List<Label> labels;
527
528                        public Item(VCard vcard, List<Label> labels) {
529                                this.vcard = vcard;
530                                this.labels = labels;
531                        }
532                }
533        }

535        /**
536         * Closes the underlying {@link Reader} object.
537         */
538        public void close() throws IOException {
539                reader.close();
540        }
