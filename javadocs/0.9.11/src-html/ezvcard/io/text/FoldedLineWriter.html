041/**
042 * Automatically folds lines as they are written.
043 * @author Michael Angstadt
044 */
045public class FoldedLineWriter extends Writer {
046        private final Writer writer;
047        private int curLineLength = 0;
048        private Integer lineLength = 75;
049        private String indent = " ";
050        private String newline = "\r\n";
051
052        /**
053         * Creates a folded line writer.
054         * @param writer the writer object to wrap
055         */
056        public FoldedLineWriter(Writer writer) {
057                this.writer = writer;
058        }
059
060        /**
061         * Writes a string, followed by a newline.
062         * @param str the text to write
063         * @throws IOException if there's a problem writing to the output stream
064         */
065        public void writeln(String str) throws IOException {
066                write(str);
067                write(newline);
068        }
069
070        /**
071         * Writes a string.
072         * @param str the string to write
073         * @param quotedPrintable true to encode the string in quoted-printable
074         * encoding, false not to
075         * @param charset the character set to use when encoding into
076         * quoted-printable, or null to use the writer's character encoding (only
077         * applicable if "quotedPrintable" is set to true)
078         * @return this
079         * @throws IOException if there's a problem writing to the output stream
080         */
081        public FoldedLineWriter append(CharSequence str, boolean quotedPrintable, Charset charset) throws IOException {
082                write(str, quotedPrintable, charset);
083                return this;
084        }
085
086        /**
087         * Writes a string.
088         * @param str the string to write
089         * @param quotedPrintable true to encode the string in quoted-printable
090         * encoding, false not to
091         * @param charset the character set to use when encoding into
092         * quoted-printable, or null to use the writer's character encoding (only
093         * applicable if "quotedPrintable" is set to true)
094         * @throws IOException if there's a problem writing to the output stream
095         */
096        public void write(CharSequence str, boolean quotedPrintable, Charset charset) throws IOException {
097                write(str.toString().toCharArray(), 0, str.length(), quotedPrintable, charset);
098        }
099
100        @Override
101        public void write(char[] cbuf, int off, int len) throws IOException {
102                write(cbuf, off, len, false, null);
103        }
104
105        /**
106         * Writes a portion of an array of characters.
107         * @param cbuf the array of characters
108         * @param off the offset from which to start writing characters
109         * @param len the number of characters to write
110         * @param quotedPrintable true to encode the string in quoted-printable
111         * encoding, false not to
112         * @param charset the character set to use when encoding into
113         * quoted-printable, or null to use the writer's character encoding (only
114         * applicable if "quotedPrintable" is set to true)
115         * @throws IOException if there's a problem writing to the output stream
116         */
117        public void write(char[] cbuf, int off, int len, boolean quotedPrintable, Charset charset) throws IOException {
118                if (quotedPrintable) {
119                        if (charset == null) {
120                                charset = Charset.forName("UTF-8");
121                        }
122
123                        QuotedPrintableCodec codec = new QuotedPrintableCodec(charset.name());
124                        try {
125                                String str = new String(cbuf, off, len);
126                                String encoded = codec.encode(str);
127
128                                cbuf = encoded.toCharArray();
129                                off = 0;
130                                len = cbuf.length;
131                        } catch (EncoderException e) {
132                                /*
133                                 * Thrown if an unsupported charset is passed into the codec.
134                                 * This should never be thrown though, because we already know
135                                 * the charset is valid (a Charset object is passed into the
136                                 * method).
137                                 */
138                                throw new RuntimeException(e);
139                        }
140                }
141
142                if (lineLength == null) {
143                        /*
144                         * If line folding is disabled, then write directly to the Writer.
145                         */
146                        writer.write(cbuf, off, len);
147                        return;
148                }
149
150                int effectiveLineLength = lineLength;
151                if (quotedPrintable) {
152                        /*
153                         * Account for the "=" character that must be appended onto each
154                         * line.
155                         */
156                        effectiveLineLength -= 1;
157                }
158
159                int encodedCharPos = -1;
160                int start = off;
161                int end = off + len;
162                for (int i = start; i < end; i++) {
163                        char c = cbuf[i];
164
165                        /*
166                         * Keep track of the quoted-printable characters to prevent them
167                         * from being cut in two at a folding boundary.
168                         */
169                        if (encodedCharPos >= 0) {
170                                encodedCharPos++;
171                                if (encodedCharPos == 3) {
172                                        encodedCharPos = -1;
173                                }
174                        }
175
176                        if (c == '\n') {
177                                writer.write(cbuf, start, i - start + 1);
178                                curLineLength = 0;
179                                start = i + 1;
180                                continue;
181                        }
182
183                        if (c == '\r') {
184                                if (i == end - 1 || cbuf[i + 1] != '\n') {
185                                        writer.write(cbuf, start, i - start + 1);
186                                        curLineLength = 0;
187                                        start = i + 1;
188                                } else {
189                                        curLineLength++;
190                                }
191                                continue;
192                        }
193
194                        if (c == '=' && quotedPrintable) {
195                                encodedCharPos = 0;
196                        }
197
198                        if (curLineLength >= effectiveLineLength) {
199                                /*
200                                 * If the last characters on the line are whitespace, then
201                                 * exceed the max line length in order to include the whitespace
202                                 * on the same line.
203                                 * 
204                                 * This is to prevent the whitespace from merging with the
205                                 * padding whitespace of the following folded line and
206                                 * potentially being lost. 2.1 allows multiple whitespace
207                                 * characters to be used for folding, so it could get lost here.
208                                 * 3.0 and 4.0 only allow one character to be used, so it
209                                 * shouldn't be lost for these versions, but better safe than
210                                 * sorry.
211                                 */
212                                if (Character.isWhitespace(c)) {
213                                        while (Character.isWhitespace(c) && i < end - 1) {
214                                                i++;
215                                                c = cbuf[i];
216                                        }
217                                        if (i >= end - 1) {
218                                                /*
219                                                 * The rest of the char array is whitespace, so leave
220                                                 * the loop.
221                                                 */
222                                                break;
223                                        }
224                                }
225
226                                /*
227                                 * If we are in the middle of a quoted-printable encoded
228                                 * character, then exceed the max line length so the sequence
229                                 * doesn't get split up across multiple lines.
230                                 */
231                                if (encodedCharPos > 0) {
232                                        i += 3 - encodedCharPos;
233                                        if (i >= end - 1) {
234                                                /*
235                                                 * The rest of the char array was a quoted-printable
236                                                 * encoded char, so leave the loop.
237                                                 */
238                                                break;
239                                        }
240                                }
241
242                                /*
243                                 * If the last char is the low (second) char in a surrogate
244                                 * pair, don't split the pair across two lines.
245                                 */
246                                if (Character.isLowSurrogate(c)) {
247                                        i++;
248                                        if (i >= end - 1) {
249                                                /*
250                                                 * Surrogate pair finishes the char array, so leave the
251                                                 * loop.
252                                                 */
253                                                break;
254                                        }
255                                }
256
257                                writer.write(cbuf, start, i - start);
258                                if (quotedPrintable) {
259                                        writer.write('=');
260                                }
261                                writer.write(newline);
262                                writer.write(indent);
263                                curLineLength = indent.length() + 1;
264                                start = i;
265
266                                continue;
267                        }
268
269                        curLineLength++;
270                }
271
272                writer.write(cbuf, start, end - start);
273        }
274
275        /**
276         * Closes the writer.
277         */
278        @Override
279        public void close() throws IOException {
280                writer.close();
281        }
282
283        /**
284         * Flushes the writer.
285         */
286        @Override
287        public void flush() throws IOException {
288                writer.flush();
289        }
290
291        /**
292         * Gets the maximum length a line can be before it is folded (excluding the
293         * newline, defaults to 75).
294         * @return the line length or null if folding is disabled
295         */
296        public Integer getLineLength() {
297                return lineLength;
298        }
299
300        /**
301         * Sets the maximum length a line can be before it is folded (excluding the
302         * newline, defaults to 75).
303         * @param lineLength the line length or null to disable folding
304         * @throws IllegalArgumentException if the line length is less than or equal
305         * to zero
306         */
307        public void setLineLength(Integer lineLength) {
308                if (lineLength != null && lineLength <= 0) {
309                        throw Messages.INSTANCE.getIllegalArgumentException(2);
310                }
311                this.lineLength = lineLength;
312        }
313
314        /**
315         * Gets the string that is prepended to each folded line (defaults to a
316         * single space character).
317         * @return the indent string
318         */
319        public String getIndent() {
320                return indent;
321        }
322
323        /**
324         * Sets the string that is prepended to each folded line (defaults to a
325         * single space character).
326         * @param indent the indent string
327         * @throws IllegalArgumentException if the length of the indent string is
328         * greater than the max line length
329         */
330        public void setIndent(String indent) {
331                if (lineLength != null && indent.length() >= lineLength) {
332                        throw Messages.INSTANCE.getIllegalArgumentException(3);
333                }
334                this.indent = indent;
335        }
336
337        /**
338         * Gets the newline sequence that is used to separate lines (defaults to
339         * CRLF).
340         * @return the newline sequence
341         */
342        public String getNewline() {
343                return newline;
344        }
345
346        /**
347         * Sets the newline sequence that is used to separate lines (defaults to
348         * CRLF).
349         * @param newline the newline sequence
350         */
351        public void setNewline(String newline) {
352                this.newline = newline;
353        }
354
355        /**
356         * Gets the wrapped {@link Writer} object.
357         * @return the wrapped writer
358         */
359        public Writer getWriter() {
360                return writer;
361        }
362
363        /**
364         * Gets the writer's character encoding.
365         * @return the writer's character encoding or null if undefined
366         */
367        public Charset getEncoding() {
368                if (!(writer instanceof OutputStreamWriter)) {
369                        return null;
370                }
371
372                OutputStreamWriter osw = (OutputStreamWriter) writer;
373                String charsetStr = osw.getEncoding();
374                return (charsetStr == null) ? null : Charset.forName(charsetStr);
375        }
