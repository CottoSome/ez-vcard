049/**
050 * Writes data to an vCard data stream.
051 * @author Michael Angstadt
052 * @see <a href="http://www.imc.org/pdi/vcard-21.rtf">vCard 2.1</a>
053 * @see <a href="http://tools.ietf.org/html/rfc2426">RFC 2426 (3.0)</a>
054 * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 (4.0)</a>
055 */
056public class VCardRawWriter implements Closeable, Flushable {
057        /**
058         * If any of these characters are found within a parameter value, then the
059         * entire parameter value must be wrapped in double quotes (applies to
060         * versions 3.0 and 4.0 only).
061         */
062        private final CharacterBitSet specialParameterCharacters = new CharacterBitSet(",:;");
063
064        /**
065         * Regular expression used to detect newline character sequences.
066         */
067        private final Pattern newlineRegex = Pattern.compile("\\r\\n|\\r|\\n");
068        private final CharacterBitSet newlineBitSet = new CharacterBitSet("\r\n");
069
070        /**
071         * List of characters which would break the syntax of the vCard if used
072         * inside a property name. The list of characters permitted by the
073         * specification is much more strict, but the goal here is to be as lenient
074         * as possible.
075         */
076        private final CharacterBitSet invalidPropertyNameCharacters = new CharacterBitSet(".;:\n\r");
077
078        /**
079         * List of characters which would break the syntax of the vCard if used
080         * inside a group name. The list of characters permitted by the
081         * specification is much more strict, but the goal here is to be as lenient
082         * as possible.
083         */
084        private final CharacterBitSet invalidGroupNameCharacters = invalidPropertyNameCharacters;
085
086        /**
087         * List of characters which would break the syntax of the vCard if used
088         * inside a parameter name. The list of characters permitted by the
089         * specification is much more strict, but the goal here is to be as lenient
090         * as possible.
091         */
092        private final CharacterBitSet invalidParameterNameCharacters = new CharacterBitSet(";:=\n\r");
093
094        /**
095         * List of characters which would break the syntax of the vCard if used
096         * inside a parameter value with caret encoding disabled. These characters
097         * cannot be escaped or encoded, so they are impossible to include inside of
098         * a parameter value. The list of characters permitted by the specification
099         * is much more strict, but the goal here is to be as lenient as possible.
100         */
101        private final Map<VCardVersion, CharacterBitSet> invalidParamValueChars;
102        {
103                Map<VCardVersion, CharacterBitSet> map = new EnumMap<VCardVersion, CharacterBitSet>(VCardVersion.class);
104
105                map.put(VCardVersion.V2_1, new CharacterBitSet(",:\n\r")); //note: semicolons can be escaped
106                map.put(VCardVersion.V3_0, new CharacterBitSet("\"\r\n"));
107                map.put(VCardVersion.V4_0, new CharacterBitSet("\""));
108
109                invalidParamValueChars = Collections.unmodifiableMap(map);
110        }
111
112        /**
113         * List of characters which would break the syntax of the vCard if used
114         * inside a parameter value with caret encoding enabled. These characters
115         * cannot be escaped or encoded, so they are impossible to include inside of
116         * a parameter value. The list of characters permitted by the specification
117         * is much more strict, but the goal here is to be as lenient as possible.
118         */
119        private final Map<VCardVersion, CharacterBitSet> invalidParamValueCharsWithCaretEncoding;
120        {
121                Map<VCardVersion, CharacterBitSet> map = new EnumMap<VCardVersion, CharacterBitSet>(VCardVersion.class);
122
123                map.put(VCardVersion.V2_1, invalidParamValueChars.get(VCardVersion.V2_1)); //2.1 does not support caret encoding
124                map.put(VCardVersion.V3_0, new CharacterBitSet(""));
125                map.put(VCardVersion.V4_0, new CharacterBitSet(""));
126
127                invalidParamValueCharsWithCaretEncoding = Collections.unmodifiableMap(map);
128        }

131        private boolean caretEncodingEnabled = false;
132        private VCardVersion version;

151        /**
152         * <p>
153         * Gets whether the writer will apply circumflex accent encoding on
154         * parameter values (disabled by default, only applies to 3.0 and 4.0
155         * vCards). This escaping mechanism allows for newlines and double quotes to
156         * be included in parameter values.
157         * </p>
158         * 
159         * <p>
160         * Note that this encoding mechanism is defined separately from the vCard
161         * specification and may not be supported by the vCard consumer.
162         * </p>
163         * 
164         * <table class="simpleTable">
165         * <caption>Characters encoded by circumflex accent encoding</caption>
166         * <tr>
167         * <th>Raw Character</th>
168         * <th>Encoded Character</th>
169         * </tr>
170         * <tr>
171         * <td>{@code "}</td>
172         * <td>{@code ^'}</td>
173         * </tr>
174         * <tr>
175         * <td><i>newline</i></td>
176         * <td>{@code ^n}</td>
177         * </tr>
178         * <tr>
179         * <td>{@code ^}</td>
180         * <td>{@code ^^}</td>
181         * </tr>
182         * </table>
183         * 
184         * <p>
185         * Example:
186         * </p>
187         * 
188         * <pre>
189         * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPittsburgh, PA 15212":40.446816;80.00566
190         * </pre>
191         * 
192         * @return true if circumflex accent encoding is enabled, false if not
193         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
194         */
195        public boolean isCaretEncodingEnabled() {
196                return caretEncodingEnabled;
197        }

205         * </p>
206         * 
207         * <p>
208         * Note that this encoding mechanism is defined separately from the vCard
209         * specification and may not be supported by the vCard consumer.
210         * </p>
211         * 
212         * <table class="simpleTable">
213         * <caption>Characters encoded by circumflex accent encoding</caption>
214         * <tr>
215         * <th>Raw Character</th>
216         * <th>Encoded Character</th>

290        /**
291         * Writes a property to the vCard data stream.
292         * @param propertyName the property name (e.g. "FN")
293         * @param value the property value
294         * @throws IllegalArgumentException if the property data contains one or
295         * more characters which break the vCard syntax and which cannot be escaped
296         * or encoded
297         * @throws IOException if there's a problem writing to the data stream
298         */
299        public void writeProperty(String propertyName, String value) throws IOException {
300                writeProperty(null, propertyName, new VCardParameters(), value);
301        }

303        /**
304         * Writes a property to the vCard data stream.
305         * @param group the group or null if there is no group
306         * @param propertyName the property name (e.g. "FN")
307         * @param parameters the property parameters
308         * @param value the property value (will be converted to "quoted-printable"
309         * encoding if the {@link Encoding#QUOTED_PRINTABLE} parameter is set)
310         * @throws IllegalArgumentException if the property data contains one or
311         * more characters which break the vCard syntax and which cannot be escaped
312         * or encoded
313         * @throws IOException if there's a problem writing to the data stream
314         */
315        public void writeProperty(String group, String propertyName, VCardParameters parameters, String value) throws IOException {
316                //validate the group name
317                if (group != null) {
318                        if (invalidGroupNameCharacters.containsAny(group)) {
319                                throw Messages.INSTANCE.getIllegalArgumentException(7, propertyName, group, printableCharacterList(invalidGroupNameCharacters.characters()));
320                        }
321                        if (beginsWithWhitespace(group)) {
322                                throw Messages.INSTANCE.getIllegalArgumentException(8, propertyName, group);
323                        }
324                }
325
326                //validate the property name
327                if (invalidPropertyNameCharacters.containsAny(propertyName)) {
328                        throw Messages.INSTANCE.getIllegalArgumentException(9, propertyName, printableCharacterList(invalidPropertyNameCharacters.characters()));
329                }
330                if (beginsWithWhitespace(propertyName)) {
331                        throw Messages.INSTANCE.getIllegalArgumentException(10, propertyName);
332                }
333
334                value = sanitizePropertyValue(value, parameters);
335
336                /*
337                 * Determine if the property value must be encoded in quoted printable
338                 * encoding. If so, then determine what charset to use for the encoding.
339                 */
340                boolean useQuotedPrintable = (parameters.getEncoding() == Encoding.QUOTED_PRINTABLE);
341                Charset quotedPrintableCharset = null;
342                if (useQuotedPrintable) {
343                        String charsetParam = parameters.getCharset();
344                        if (charsetParam == null) {
345                                quotedPrintableCharset = Charset.forName("UTF-8");
346                        } else {
347                                try {
348                                        quotedPrintableCharset = Charset.forName(charsetParam);
349                                } catch (Throwable t) {
350                                        quotedPrintableCharset = Charset.forName("UTF-8");
351                                }
352                        }
353                        parameters.setCharset(quotedPrintableCharset.name());
354                }
355
356                //write the group
357                if (group != null) {
358                        writer.append(group).append('.');
359                }
360
361                //write the property name
362                writer.append(propertyName);
363
364                //write the parameters
365                for (Map.Entry<String, List<String>> parameter : parameters) {
366                        String parameterName = parameter.getKey();
367                        List<String> parameterValues = parameter.getValue();
368                        if (parameterValues.isEmpty()) {
369                                continue;
370                        }
371
372                        //check the parameter name for invalid characters
373                        if (invalidParameterNameCharacters.containsAny(parameterName)) {
374                                throw Messages.INSTANCE.getIllegalArgumentException(11, propertyName, parameterName, printableCharacterList(invalidParameterNameCharacters.characters()));
375                        }
376
377                        if (version == VCardVersion.V2_1) {
378                                boolean isTypeParameter = VCardParameters.TYPE.equalsIgnoreCase(parameterName);
379                                for (String parameterValue : parameterValues) {
380                                        parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
381
382                                        if (isTypeParameter) {
383                                                //e.g. ADR;HOME;WORK:
384                                                writer.append(';').append(parameterValue.toUpperCase());
385                                        } else {
386                                                //e.g. ADR;FOO=bar;FOO=car:
387                                                writer.append(';').append(parameterName).append('=').append(parameterValue);
388                                        }
389                                }
390                                continue;
391                        }
392
393                        //e.g. ADR;TYPE=home,work,"another,value":
394                        boolean first = true;
395                        writer.append(';').append(parameterName).append('=');
396                        for (String parameterValue : parameterValues) {
397                                if (!first) {
398                                        writer.append(',');
399                                }
400
401                                parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
402                                if (specialParameterCharacters.containsAny(parameterValue)) {
403                                        writer.append('"').append(parameterValue).append('"');
404                                } else {
405                                        writer.append(parameterValue);
406                                }
407
408                                first = false;
409                        }
410                }
411
412                writer.append(':');
413                writer.append(value, useQuotedPrintable, quotedPrintableCharset);
414                writer.append(writer.getNewline());
415        }

417        /**
418         * Determines if a given string starts with whitespace.
419         * @param string the string
420         * @return true if it starts with whitespace, false if not
421         */
422        private boolean beginsWithWhitespace(String string) {
423                if (string.length() == 0) {
424                        return false;
425                }
426                char first = string.charAt(0);
427                return (first == ' ' || first == '\t');
428        }

430        /**
431         * Sanitizes a property value for safe inclusion in a vCard.
432         * @param value the value to sanitize
433         * @param parameters the property's parameters
434         * @return the sanitized value
435         */
436        private String sanitizePropertyValue(String value, VCardParameters parameters) {
437                if (value == null) {
438                        return "";
439                }
440
441                /*
442                 * 2.1 does not support the "\n" escape sequence (see "Delimiters"
443                 * sub-section in section 2 of the specs) so encode the value in
444                 * quoted-printable encoding if any newline characters exist.
445                 */
446                if (version == VCardVersion.V2_1 && newlineBitSet.containsAny(value)) {
447                        parameters.setEncoding(Encoding.QUOTED_PRINTABLE);
448                        return value;
449                }
450
451                return escapeNewlines(value);
452        }
453
454        /**
455         * Sanitizes a parameter value for serialization.
456         * @param parameterValue the parameter value
457         * @param parameterName the parameter name
458         * @param propertyName the name of the property to which the parameter
459         * belongs
460         * @return the sanitized parameter value
461         * @throws IllegalArgumentException if the value contains invalid characters
462         */
463        private String sanitizeParameterValue(String parameterValue, String parameterName, String propertyName) {
464                CharacterBitSet invalidChars = (caretEncodingEnabled ? invalidParamValueCharsWithCaretEncoding : invalidParamValueChars).get(version);
465                if (invalidChars.containsAny(parameterValue)) {
466                        throw Messages.INSTANCE.getIllegalArgumentException(12, propertyName, parameterName, printableCharacterList(invalidChars.characters()));
467                }
468
469                String sanitizedValue = parameterValue;
470                switch (version) {
471                case V2_1:
472                        //Note: 2.1 does not support caret encoding.
473
474                        //escape backslashes
475                        sanitizedValue = sanitizedValue.replace("\\", "\\\\");
476
477                        //escape semi-colons (see section 2)
478                        return sanitizedValue.replace(";", "\\;");
479
480                case V3_0:
481                        if (caretEncodingEnabled) {
482                                return applyCaretEncoding(sanitizedValue);
483                        }
484                        return sanitizedValue;
485
486                case V4_0:
487                        if (caretEncodingEnabled) {
488                                return applyCaretEncoding(sanitizedValue);
489                        }
490
491                        /*
492                         * 4.0 allows newlines to be escaped (for the LABEL parameter).
493                         */
494                        return newlineRegex.matcher(sanitizedValue).replaceAll("\\\\\\n");
495                }
496
497                return "";
498        }
499
500        private String printableCharacterList(String list) {
501                return list.replace("\n", "\\n").replace("\r", "\\r");
502        }
503
504        /**
505         * Applies circumflex accent encoding to a string.
506         * @param value the string
507         * @return the encoded string
508         */
509        private String applyCaretEncoding(String value) {
510                value = value.replace("^", "^^");
511                value = newlineRegex.matcher(value).replaceAll("^n");
512                value = value.replace("\"", "^'");
513                return value;
514        }
515
516        /**
517         * Escapes all newlines in a string.
518         * @param string the string to escape
519         * @return the escaped string
520         */
521        private String escapeNewlines(String string) {
522                return newlineRegex.matcher(string).replaceAll("\\\\n");
523        }
