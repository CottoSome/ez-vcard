059/**
060 * <p>
061 * Writes {@link VCard} objects to a plain-text vCard data stream.
062 * </p>
063 * <p>
064 * <b>Example:</b>
065 * </p>
066 * 
067 * <pre class="brush:java">
068 * VCard vcard1 = ...
069 * VCard vcard2 = ...
070 * File file = new File("vcard.vcf");
071 * VCardWriter writer = null;
072 * try {
073 *   writer = new VCardWriter(file, VCardVersion.V3_0);
074 *   writer.write(vcard1);
075 *   writer.write(vcard2);
076 * } finally {
077 *   if (writer != null) writer.close();
078 * }
079 * </pre>
080 * 
081 * <p>
082 * <b>Changing the line folding settings:</b>
083 * </p>
084 * 
085 * <pre class="brush:java">
086 * VCardWriter writer = new VCardWriter(...);
087 * 
088 * //disable line folding
089 * writer.getRawWriter().getFoldedLineWriter().setLineLength(null);
090 * 
091 * //change line length
092 * writer.getRawWriter().getFoldedLineWriter().setLineLength(50);
093 * 
094 * //change folded line indent string
095 * writer.getRawWriter().getFoldedLineWriter().setIndent("\t");
096 * 
097 * //change newline character
098 * writer.getRawWriter().getFoldedLineWriter().setNewline("**");
099 * </pre>
100 * @author Michael Angstadt
101 * @see <a href="http://www.imc.org/pdi/vcard-21.rtf">vCard 2.1</a>
102 * @see <a href="http://tools.ietf.org/html/rfc2426">RFC 2426 (3.0)</a>
103 * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 (4.0)</a>
104 */
105public class VCardWriter extends StreamWriter implements Flushable {
106        private final VCardRawWriter writer;
107        private final LinkedList<Boolean> prodIdStack = new LinkedList<Boolean>();
108        private TargetApplication targetApplication;
109        private Boolean includeTrailingSemicolons;

111        /**
112         * @param out the output stream to write to
113         * @param targetVersion the version that the vCards should conform to (if
114         * set to "4.0", vCards will be written in UTF-8 encoding)
115         */
116        public VCardWriter(OutputStream out, VCardVersion targetVersion) {
117                this((targetVersion == VCardVersion.V4_0) ? utf8Writer(out) : new OutputStreamWriter(out), targetVersion);
118        }

125         */
126        public VCardWriter(File file, VCardVersion targetVersion) throws IOException {
127                this(file, false, targetVersion);
128        }

287        @Override
288        @SuppressWarnings({ "rawtypes", "unchecked" })
289        protected void _write(VCard vcard, List<VCardProperty> propertiesToAdd) throws IOException {
290                VCardVersion targetVersion = getTargetVersion();
291
292                Boolean includeTrailingSemicolons = this.includeTrailingSemicolons;
293                if (includeTrailingSemicolons == null) {
294                        includeTrailingSemicolons = (targetVersion == VCardVersion.V4_0);
295                }
296
297                WriteContext context = new WriteContext(targetVersion, getTargetApplication(), includeTrailingSemicolons);
298
299                writer.writeBeginComponent("VCARD");
300                writer.writeVersion();
301
302                for (VCardProperty property : propertiesToAdd) {
303                        VCardPropertyScribe scribe = index.getPropertyScribe(property);
304
305                        //marshal the value
306                        String value = null;
307                        VCard nestedVCard = null;
308                        try {
309                                value = scribe.writeText(property, context);
310                        } catch (SkipMeException e) {
311                                continue;
312                        } catch (EmbeddedVCardException e) {
313                                nestedVCard = e.getVCard();
314                        }
315
316                        //marshal the parameters
317                        VCardParameters parameters = scribe.prepareParameters(property, targetVersion, vcard);
318
319                        //is the value a nested vCard?
320                        if (nestedVCard != null) {
321                                if (targetVersion == VCardVersion.V2_1) {
322                                        //write a nested vCard (2.1 style)
323                                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
324                                        prodIdStack.add(addProdId);
325                                        addProdId = false;
326                                        write(nestedVCard);
327                                        addProdId = prodIdStack.removeLast();
328                                } else {
329                                        //write an embedded vCard (3.0 style)
330                                        StringWriter sw = new StringWriter();
331                                        VCardWriter agentWriter = new VCardWriter(sw, targetVersion);
332                                        agentWriter.getRawWriter().getFoldedLineWriter().setLineLength(null);
333                                        agentWriter.getRawWriter().getFoldedLineWriter().setNewline("\n");
334                                        agentWriter.setAddProdId(false);
335                                        agentWriter.setIncludeTrailingSemicolons(includeTrailingSemicolons);
336                                        agentWriter.setTargetApplication(getTargetApplication());
337                                        agentWriter.setVersionStrict(versionStrict);
338                                        try {
339                                                agentWriter.write(nestedVCard);
340                                        } catch (IOException e) {
341                                                //should never be thrown because we're writing to a string
342                                        } finally {
343                                                IOUtils.closeQuietly(agentWriter);
344                                        }
345
346                                        String vcardStr = sw.toString();
347                                        vcardStr = VCardPropertyScribe.escape(vcardStr);
348                                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, vcardStr);
349                                }
350                                continue;
351                        }
352
353                        /*
354                         * Set the property's data type.
355                         * 
356                         * Only add a VALUE parameter if the data type is: (1) not "unknown"
357                         * (2) different from the property's default data type (3) not the
358                         * date/time special case (see code)
359                         */
360                        VCardDataType dataType = scribe.dataType(property, targetVersion);
361                        if (dataType != null) {
362                                VCardDataType defaultDataType = scribe.defaultDataType(targetVersion);
363                                if (dataType != defaultDataType && !isDateTimeValueParameterSpecialCase(defaultDataType, dataType)) {
364                                        parameters.setValue(dataType);
365                                }
366                        }
367
368                        //write the property
369                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
370
371                        fixBinaryPropertyForOutlook(property);
372                }
373
374                writer.writeEndComponent("VCARD");
375        }
376
377        /**
378         * @see TargetApplication#OUTLOOK
379         */
380        private void fixBinaryPropertyForOutlook(VCardProperty property) throws IOException {
381                if (targetApplication != TargetApplication.OUTLOOK) {
382                        return;
383                }
384
385                if (getTargetVersion() == VCardVersion.V4_0) {
386                        //only do this for 2.1 and 3.0 vCards
387                        return;
388                }
389
390                if (!(property instanceof BinaryProperty)) {
391                        //property does not have binary data
392                        return;
393                }
394
395                BinaryProperty<?> binaryProperty = (BinaryProperty<?>) property;
396                if (binaryProperty.getData() == null) {
397                        //property value is not base64-encoded
398                        return;
399                }
400
401                writer.getFoldedLineWriter().writeln("");
402        }
403
404        /**
405         * Determines if the given default data type is "date-and-or-time" and the
406         * given data type is time-based. Properties that meet this criteria should
407         * NOT be given a VALUE parameter.
408         * @param defaultDataType the property's default data type
409         * @param dataType the current property instance's data type
410         * @return true if the default data type is "date-and-or-time" and the data
411         * type is time-based, false otherwise
412         */
413        private boolean isDateTimeValueParameterSpecialCase(VCardDataType defaultDataType, VCardDataType dataType) {
414                //@formatter:off
415                return
416                defaultDataType == VCardDataType.DATE_AND_OR_TIME &&
417                (
418                        dataType == VCardDataType.DATE ||
419                        dataType == VCardDataType.DATE_TIME ||
420                        dataType == VCardDataType.TIME
421                );
422                //@formatter:on
423        }
